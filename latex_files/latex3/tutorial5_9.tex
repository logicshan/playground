% int do 循环
\documentclass{article}
\begin{document}
\ExplSyntaxOn
\int_new:N \l_i_int
\int_new:N \l_j_int
\int_set:Nn \l_i_int { 0 }
\int_set:Nn \l_j_int { 10 }
\int_do_while:nNnn
  { \l_i_int } < { \l_j_int }
  { 
    \int_use:N \l_i_int < 
    \int_use:N \l_j_int \par
    \int_incr:N \l_i_int
    \int_decr:N \l_j_int
  }
\ExplSyntaxOff
\end{document}
